400      * @param c  the outer constraint ({@code null} not permitted).

404     protected RectangleConstraint toContentConstraint(RectangleConstraint c) {
405         Args.nullNotPermitted(c, "c");
406         if (c.equals(RectangleConstraint.NONE)) {
407             return c;
408         }
409         double w = c.getWidth();
410         Range wr = c.getWidthRange();
411         double h = c.getHeight();
412         Range hr = c.getHeightRange();
413         double ww = trimToContentWidth(w);
414         double hh = trimToContentHeight(h);
415         Range wwr = trimToContentWidth(wr);
416         Range hhr = trimToContentHeight(hr);
417         return new RectangleConstraint(ww, wwr, c.getWidthConstraintType(),
418             hh, hhr, c.getHeightConstraintType());
419     }

421     private Range trimToContentWidth(Range r) {
422         if (r == null) {
423             return null;
424         }
425         double lowerBound = 0.0;
426         double upperBound = Double.POSITIVE_INFINITY;
427         if (r.getLowerBound() > 0.0) {
428             lowerBound = trimToContentWidth(r.getLowerBound());
429         }
430         if (r.getUpperBound() < Double.POSITIVE_INFINITY) {
431             upperBound = trimToContentWidth(r.getUpperBound());
432         }
433         return new Range(lowerBound, upperBound);
434     }
