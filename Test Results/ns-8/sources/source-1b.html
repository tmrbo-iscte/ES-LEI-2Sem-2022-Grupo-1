59     /**
60      * Generates a tool tip text item for a particular item within a series.
61      *
62      * @param dataset  the dataset ({@code null} not permitted).
63      * @param series  the series number (zero-based index).
64      * @param item  the item number (zero-based index).
65      *
66      * @return The tool tip text (possibly {@code null}).
67      */
68     @Override
69     public String generateToolTip(XYDataset dataset, int series, int item) {
70         Args.nullNotPermitted(dataset, "dataset");
71         String xStr, yStr;
72         if (dataset instanceof YisSymbolic) {
73             yStr = ((YisSymbolic) dataset).getYSymbolicValue(series, item);
74         }
75         else {
76             double y = dataset.getYValue(series, item);
77             yStr = Double.toString(round(y, 2));
78         }
79         if (dataset instanceof XisSymbolic) {
80             xStr = ((XisSymbolic) dataset).getXSymbolicValue(series, item);
81         }
82         else if (dataset instanceof TimeSeriesCollection) {
83             RegularTimePeriod p
84                 = ((TimeSeriesCollection) dataset).getSeries(series)
85                     .getTimePeriod(item);
86             xStr = p.toString();
87         }
88         else {
89             double x = dataset.getXValue(series, item);
90             xStr = Double.toString(round(x, 2));
91         }
92         return "X: " + xStr + ", Y: " + yStr;
93     }
