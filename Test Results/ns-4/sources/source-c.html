83     /**
84      * Arranges the blocks within the specified container, subject to the given
85      * constraint.
86      *
87      * @param container  the container ({@code null} not permitted).
88      * @param constraint  the constraint.
89      * @param g2  the graphics device.
90      *
91      * @return The size following the arrangement.
92      */
93     @Override
94     public Size2D arrange(BlockContainer container, Graphics2D g2,
95                           RectangleConstraint constraint) {
96         LengthConstraintType w = constraint.getWidthConstraintType();
97         LengthConstraintType h = constraint.getHeightConstraintType();
98         if (w == LengthConstraintType.NONE) {
99             if (h == LengthConstraintType.NONE) {
100                 return arrangeNN(container, g2);
101             }
102             else if (h == LengthConstraintType.FIXED) {
103                 return arrangeNF(container, g2, constraint);
104             }
105             else if (h == LengthConstraintType.RANGE) {
106                 // find optimum height, then map to range
107                 return arrangeNR(container, g2, constraint);
108             }
109         }
110         else if (w == LengthConstraintType.FIXED) {
111             if (h == LengthConstraintType.NONE) {
112                 // find optimum height
113                 return arrangeFN(container, g2, constraint);
114             }
115             else if (h == LengthConstraintType.FIXED) {
116                 return arrangeFF(container, g2, constraint);
117             }
118             else if (h == LengthConstraintType.RANGE) {
119                 // find optimum height and map to range
120                 return arrangeFR(container, g2, constraint);
121             }
122         }
123         else if (w == LengthConstraintType.RANGE) {
124             // find optimum width and map to range
125             if (h == LengthConstraintType.NONE) {
126                 // find optimum height
127                 return arrangeRN(container, g2, constraint);
128             }
129             else if (h == LengthConstraintType.FIXED) {
130                 // fixed width
131                 return arrangeRF(container, g2, constraint);
132             }
133             else if (h == LengthConstraintType.RANGE) {
134                 return arrangeRR(container, g2, constraint);
135             }
136         }
137         throw new RuntimeException("Should never get to here!");
138     }

140     /**
141      * Arranges the container with no constraint on the width or height.
142      *
143      * @param container  the container ({@code null} not permitted).
144      * @param g2  the graphics device.
145      *
146      * @return The size.
147      */
148     protected Size2D arrangeNN(BlockContainer container, Graphics2D g2) {
149         double maxW = 0.0;
150         double maxH = 0.0;
151         List<Block> blocks = container.getBlocks();
152         for (Block b : blocks) {
153             if (b != null) {
154                 Size2D s = b.arrange(g2, RectangleConstraint.NONE);
155                 maxW = Math.max(maxW, s.width);
156                 maxH = Math.max(maxH, s.height);
157             }
158         }
159         double width = this.columns * maxW;
160         double height = this.rows * maxH;
161         RectangleConstraint c = new RectangleConstraint(width, height);
162         return arrangeFF(container, g2, c);
163     }

165     /**
166      * Arranges the container with a fixed overall width and height.
167      *
168      * @param container  the container ({@code null} not permitted).
169      * @param g2  the graphics device.
170      * @param constraint  the constraint ({@code null} not permitted).
171      *
172      * @return The size following the arrangement.
173      */
174     protected Size2D arrangeFF(BlockContainer container, Graphics2D g2,
175                                RectangleConstraint constraint) {
176         double width = constraint.getWidth() / this.columns;
177         double height = constraint.getHeight() / this.rows;
178         List<Block> blocks = container.getBlocks();
179         for (int c = 0; c < this.columns; c++) {
180             for (int r = 0; r < this.rows; r++) {
181                 int index = r * this.columns + c;
182                 if (index >= blocks.size()) {
183                     break;
184                 }
185                 Block b = blocks.get(index);
186                 if (b != null) {
187                     b.setBounds(new Rectangle2D.Double(c * width, r * height,
188                             width, height));
189                 }
190             }
191         }
192         return new Size2D(this.columns * width, this.rows * height);
193     }

204     protected Size2D arrangeFR(BlockContainer container, Graphics2D g2,
205                                RectangleConstraint constraint) {
206 
207         RectangleConstraint c1 = constraint.toUnconstrainedHeight();
208         Size2D size1 = arrange(container, g2, c1);
209 
210         if (constraint.getHeightRange().contains(size1.getHeight())) {
211             return size1;
212         }
213         else {
214             double h = constraint.getHeightRange().constrain(size1.getHeight());
215             RectangleConstraint c2 = constraint.toFixedHeight(h);
216             return arrange(container, g2, c2);
217         }

306 
307         Size2D size1 = arrange(container, g2, RectangleConstraint.NONE);
308 
309         if (constraint.getWidthRange().contains(size1.getWidth())) {
310             if (constraint.getHeightRange().contains(size1.getHeight())) {
311                 return size1;
312             }

320             }
321         }
322         else {
323             if (constraint.getHeightRange().contains(size1.getHeight())) {
324                 // height is OK, but width must be constrained
325                 double w = constraint.getWidthRange().constrain(
326                         size1.getWidth());
327                 RectangleConstraint cc = new RectangleConstraint(w,
328                         size1.getHeight());
329                 return arrangeFF(container, g2, cc);

358         double maxH = 0.0;

363                     break;
