41 /**
42  * An object that is responsible for arranging a collection of {@link Block}s
43  * within a {@link BlockContainer}.
44  */
45 public interface Arrangement {
46 
47     /**
48      * Adds a block and a key which can be used to determine the position of
49      * the block in the arrangement.  This method is called by the container
50      * (you don't need to call this method directly) and gives the arrangement
51      * an opportunity to record the details if they are required.
52      *
53      * @param block  the block.
54      * @param key  the key ({@code null} permitted).
55      */
56     void add(Block block, Object key);
57 
58     /**
59      * Arranges the blocks within the specified container, subject to the given
60      * constraint.
61      *
62      * @param container  the container ({@code null} not permitted).
63      * @param g2  the graphics device.
64      * @param constraint  the constraint.
65      *
66      * @return The container size after the arrangement.
67      */
68     Size2D arrange(BlockContainer container, Graphics2D g2, RectangleConstraint constraint);
69 
70     /**
71      * Clears any cached layout information retained by the arrangement.
72      */
73     void clear();
74 
75 }
