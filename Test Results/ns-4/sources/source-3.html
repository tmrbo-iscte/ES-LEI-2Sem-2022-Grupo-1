179     /**
180      * Draws the block within the specified area.
181      *
182      * @param g2  the graphics device.
183      * @param area  the area.
184      * @param params  passed on to blocks within the container
185      *                ({@code null} permitted).
186      *
187      * @return An instance of {@link EntityBlockResult}, or {@code null}.
188      */
189     @Override
190     public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
191         // check if we need to collect chart entities from the container
192         EntityBlockParams ebp;
193         StandardEntityCollection sec = null;
194         if (params instanceof EntityBlockParams) {
195             ebp = (EntityBlockParams) params;
196             if (ebp.getGenerateEntities()) {
197                 sec = new StandardEntityCollection();
198             }
199         }
200         Rectangle2D contentArea = (Rectangle2D) area.clone();
201         contentArea = trimMargin(contentArea);
202         drawBorder(g2, contentArea);
203         contentArea = trimBorder(contentArea);
204         contentArea = trimPadding(contentArea);
205         for (Block block : this.blocks) {
206             Rectangle2D bounds = block.getBounds();
207             Rectangle2D drawArea = new Rectangle2D.Double(bounds.getX()
208                     + area.getX(), bounds.getY() + area.getY(),
209                     bounds.getWidth(), bounds.getHeight());
210             Object r = block.draw(g2, drawArea, params);
211             if (sec != null) {
212                 if (r instanceof EntityBlockResult) {
213                     EntityBlockResult ebr = (EntityBlockResult) r;
214                     EntityCollection ec = ebr.getEntityCollection();
215                     sec.addAll(ec);
216                 }
217             }
218         }
219         BlockResult result = null;
220         if (sec != null) {
221             result = new BlockResult();
222             result.setEntityCollection(sec);
223         }
224         return result;
225     }
