42 /**
43  * A source that can used by the {@link NumberAxis} class to obtain a
44  * suitable {@link TickUnit}.  Instances of this class are {@link Serializable}
45  * from version 1.0.7 onwards.  Cloning is not supported, because instances
46  * are immutable.
47  */
48 public class StandardTickUnitSource implements TickUnitSource, Serializable {
49 
50     /** Constant for log(10.0). */
51     private static final double LOG_10_VALUE = Math.log(10.0);
52 
53     /**
54      * Default constructor.
55      */
56     public StandardTickUnitSource() {
57         super();
58     }
59 
60     /**
61      * Returns a tick unit that is larger than the supplied unit.
62      *
63      * @param unit  the unit ({@code null} not permitted).
64      *
65      * @return A tick unit that is larger than the supplied unit.
66      */
67     @Override
68     public TickUnit getLargerTickUnit(TickUnit unit) {
69         double x = unit.getSize();
70         double log = Math.log(x) / LOG_10_VALUE;
71         double higher = Math.ceil(log);
72         return new NumberTickUnit(Math.pow(10, higher),
73                 new DecimalFormat("0.0E0"));
74     }
75 
76     /**
77      * Returns the tick unit in the collection that is greater than or equal
78      * to (in size) the specified unit.
79      *
80      * @param unit  the unit ({@code null} not permitted).
81      *
82      * @return A unit from the collection.
83      */
84     @Override
85     public TickUnit getCeilingTickUnit(TickUnit unit) {
86         return getLargerTickUnit(unit);
87     }
88 
89     /**
90      * Returns the tick unit in the collection that is greater than or equal
91      * to the specified size.
92      *
93      * @param size  the size.
94      *
95      * @return A unit from the collection.
96      */
97     @Override
98     public TickUnit getCeilingTickUnit(double size) {
99         double log = Math.log(size) / LOG_10_VALUE;
100         double higher = Math.ceil(log);
101         return new NumberTickUnit(Math.pow(10, higher),
102                 new DecimalFormat("0.0E0"));
103     }
