72     /**
73      * Calculates and sets the bounds of all the items in the specified
74      * container, subject to the given constraint.  The {@code Graphics2D}
75      * can be used by some items (particularly items containing text) to
76      * calculate sizing parameters.
77      *
78      * @param container  the container whose items are being arranged.
79      * @param g2  the graphics device.
80      * @param constraint  the size constraint.
81      *
82      * @return The size of the container after arrangement of the contents.
83      */
84     @Override
85     public Size2D arrange(BlockContainer container, Graphics2D g2,
86                           RectangleConstraint constraint) {
87 
88         LengthConstraintType w = constraint.getWidthConstraintType();
89         LengthConstraintType h = constraint.getHeightConstraintType();
90         if (w == LengthConstraintType.NONE) {
91             if (h == LengthConstraintType.NONE) {
92                 return arrangeNN(container, g2);
93             }
94             else if (h == LengthConstraintType.FIXED) {
95                 throw new RuntimeException("Not implemented.");
96             }
97             else if (h == LengthConstraintType.RANGE) {
98                 throw new RuntimeException("Not implemented.");
99             }
100         }
101         else if (w == LengthConstraintType.FIXED) {
102             if (h == LengthConstraintType.NONE) {
103                 return arrangeFN(container, g2, constraint);
104             }
105             else if (h == LengthConstraintType.FIXED) {
106                 throw new RuntimeException("Not implemented.");
107             }
108             else if (h == LengthConstraintType.RANGE) {
109                 throw new RuntimeException("Not implemented.");
110             }
111         }
112         else if (w == LengthConstraintType.RANGE) {
113             if (h == LengthConstraintType.NONE) {
114                 return arrangeRN(container, g2, constraint);
115             }
116             else if (h == LengthConstraintType.FIXED) {
117                 return arrangeRF(container, g2, constraint);
118             }
119             else if (h == LengthConstraintType.RANGE) {
120                 return arrangeRR(container, g2, constraint);
121             }
122         }
123         throw new IllegalArgumentException("Unknown LengthConstraintType.");
124 
125     }

127     /**
128      * Arranges the blocks in the container with a fixed width and no height
129      * constraint.
130      *
131      * @param container  the container.
132      * @param g2  the graphics device.
133      * @param constraint  the constraint.
134      *
135      * @return The size.
136      */
137     protected Size2D arrangeFN(BlockContainer container, Graphics2D g2,
138                                RectangleConstraint constraint) {
139 
140         List<Block> blocks = container.getBlocks();
141         Block b = blocks.get(0);
142         Size2D s = b.arrange(g2, RectangleConstraint.NONE);
143         double width = constraint.getWidth();
144         Rectangle2D bounds = new Rectangle2D.Double((width - s.width) / 2.0,
145                 0.0, s.width, s.height);
146         b.setBounds(bounds);
147         return new Size2D((width - s.width) / 2.0, s.height);
148     }

275     protected Size2D arrangeNN(BlockContainer container, Graphics2D g2) {
276         List<Block> blocks = container.getBlocks();
277         Block b = blocks.get(0);
278         Size2D s = b.arrange(g2, RectangleConstraint.NONE);
