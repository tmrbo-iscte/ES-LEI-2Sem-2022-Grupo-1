43 /**
44  * Some utility methods for calculating hash codes.  
45  */
46 public class HashUtils {
47     
48     /**
49      * Returns a hash code for a {@code Paint} instance.  If 
50      * {@code p} is {@code null}, this method returns zero.
51      * 
52      * @param p  the paint ({@code null} permitted).
53      * 
54      * @return The hash code.
55      */
56     public static int hashCodeForPaint(Paint p) {
57         if (p == null) {
58             return 0;
59         }
60         int result;
61         // handle GradientPaint as a special case
62         if (p instanceof GradientPaint) {
63             GradientPaint gp = (GradientPaint) p;
64             result = 193;
65             result = 37 * result + gp.getColor1().hashCode();
66             result = 37 * result + gp.getPoint1().hashCode();
67             result = 37 * result + gp.getColor2().hashCode();
68             result = 37 * result + gp.getPoint2().hashCode();
69         }
70         else {
71             // we assume that all other Paint instances implement equals() and
72             // hashCode()...of course that might not be true, but what can we
73             // do about it?
74             result = p.hashCode();
75         }
76         return result;
77     }
78     
79     /**
80      * Returns a hash code for a {@code double[]} instance.  If the array
81      * is {@code null}, this method returns zero.
82      * 
83      * @param a  the array ({@code null} permitted).
84      * 
85      * @return The hash code.
86      */
87     public static int hashCodeForDoubleArray(double[] a) {
88         if (a == null) { 
89             return 0;
90         }
91         int result = 193;
92         long temp;
93         for (double v : a) {
94             temp = Double.doubleToLongBits(v);
95             result = 29 * result + (int) (temp ^ (temp >>> 32));
96         }
97         return result;
98     }
