43 /**
44  * An interface used by the {@link DateAxis} and {@link NumberAxis} classes to
45  * obtain a suitable {@link TickUnit}.
46  */
47 public interface TickUnitSource {
48 
49     /**
50      * Returns the smallest tick unit available in the source that is larger
51      * than {@code unit} or, if there is no larger unit, returns {@code unit}.
52      *
53      * @param unit  the unit ({@code null} not permitted).
54      *
55      * @return A tick unit that is larger than the supplied unit.
56      */
57     TickUnit getLargerTickUnit(TickUnit unit);
58 
59     /**
60      * Returns the tick unit in the collection that is greater than or equal
61      * to (in size) the specified unit.
62      *
63      * @param unit  the unit.
64      *
65      * @return A unit from the collection.
66      */
67     TickUnit getCeilingTickUnit(TickUnit unit);
68 
69     /**
70      * Returns the smallest tick unit available in the source that is greater 
71      * than or equal to the specified size.  If there is no such tick unit,
72      * the method should return the largest available tick in the source.
73      *
74      * @param size  the size.
75      *
76      * @return A unit from the collection (never {@code null}).
77      */
78     TickUnit getCeilingTickUnit(double size);
79 
80 }
