52     /**
53      * Returns a copy of the specified object, which should be a clone if the
54      * object is cloneable otherwise a reference to the specified object is
55      * returned.  If the object is {@code null} this method returns {@code null}.
56      *
57      * @param object the object to copy ({@code null} permitted).
58      * 
59      * @param <T>  the object type.
60      * 
61      * @return A clone of the specified object, or {@code null}.
62      * 
63      * @throws CloneNotSupportedException if the object cannot be cloned.
64      */
65     public static <T> T copy(T object) throws CloneNotSupportedException {
66         if (object == null) {
67             return null;
68         }
69         if (object instanceof PublicCloneable) {
70             PublicCloneable pc = (PublicCloneable) object;
71             return (T) pc.clone();
72         } else {
73             try {
74                 Method method = object.getClass().getMethod("clone",
75                         (Class[]) null);
76                 if (Modifier.isPublic(method.getModifiers())) {
77                     return (T) method.invoke(object, (Object[]) null);
78                 } else {
79                     return object;
80                 }
81             } catch (NoSuchMethodException e) {
82                 return object;
83             } catch (IllegalAccessException e) {
84                 throw new CloneNotSupportedException("Object.clone(): unable to call method.");
85             } catch (InvocationTargetException e) {
86                 throw new CloneNotSupportedException("Object without clone() method is impossible.");
87             }
88         }
89     }

91     /**
92      * Returns a clone of the specified object, if it can be cloned, otherwise
93      * throws a {@code CloneNotSupportedException}.  If the object is 
94      * {@code null} this method returns {@code null}.
95      *
96      * @param object the object to clone ({@code null} permitted).
97      * 
98      * @param <T>  the object type.
99      * 
100      * @return A clone of the specified object, or {@code null}.
101      * 
102      * @throws CloneNotSupportedException if the object cannot be cloned.
103      */
104     public static <T> T clone(T object) throws CloneNotSupportedException {
105         if (object == null) {
106             return null;
107         }
108         if (object instanceof PublicCloneable) {
109             PublicCloneable pc = (PublicCloneable) object;
110             return (T) pc.clone();
111         } else {
112             try {
113                 Method method = object.getClass().getMethod("clone",
114                         (Class[]) null);
115                 if (Modifier.isPublic(method.getModifiers())) {
116                     return (T) method.invoke(object, (Object[]) null);
117                 }
118             } catch (NoSuchMethodException e) {
119                 throw new CloneNotSupportedException("Object without clone() method is impossible.");
120             } catch (IllegalAccessException e) {
121                 throw new CloneNotSupportedException("Object.clone(): unable to call method.");
122             } catch (InvocationTargetException e) {
123                 throw new CloneNotSupportedException("Object without clone() method is impossible.");
124             }
125         }
126         throw new CloneNotSupportedException("Failed to clone.");
127     }

129     /**
130      * Returns a list containing copies (clones if possible) of the items in 
131      * the source list.
132      * 
133      * @param source  the source list ({@code null} not permitted).
134      * 
135      * @param <T>  the type of the list items.
136      * 
137      * @return A new list. 
138      */
139     public static <T> List<T>cloneList(List<T> source) {
140         Args.nullNotPermitted(source, "source");
141         List<T> result = new ArrayList<>();
142         for (Object obj: source) {
143             try {
144                 result.add((T) copy(obj));
145             } catch (CloneNotSupportedException ex) {
146                 throw new RuntimeException(ex);
147             }
148         }
149         return result;
150     }

152     /**
153      * Returns a new map that contains the same keys and copies of the
154      * values from the source map.
155      * 
156      * @param source  the source map ({@code null} not permitted).
157      * 
158      * @param <K>  the type for the keys.
159      * @param <V>  the type for the values.
160      * 
161      * @return A new map. 
162      */
163     public static <K, V> Map<K, V> cloneMapValues(Map<K, V> source) {
164         Args.nullNotPermitted(source, "source");
165         Map<K, V> result = new HashMap<>();
166         for (K key : source.keySet()) {
167             V value = source.get(key);
168             if (value != null) {
169                 try {
170                     result.put(key, copy(value));
171                 } catch (CloneNotSupportedException ex) {
172                     throw new RuntimeException(ex);
173                 }
174             } else {
175                 result.put(key, null);
176             }
177         }
178         return result;
179     }
