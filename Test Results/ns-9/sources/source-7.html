408         Size2D contentSize = null;

413             else if (h == LengthConstraintType.RANGE) {
414                 throw new RuntimeException("Not yet implemented.");
415             }
416             else if (h == LengthConstraintType.FIXED) {
417                 throw new RuntimeException("Not yet implemented.");
418             }
419         }
420         else if (w == LengthConstraintType.RANGE) {
421             if (h == LengthConstraintType.NONE) {
422                 throw new RuntimeException("Not yet implemented.");
423             }
424             else if (h == LengthConstraintType.RANGE) {
425                 contentSize = arrangeRR(g2, cc.getWidthRange(),
426                         cc.getHeightRange());
427             }

449      * Returns the content size for the title.  This will reflect the fact that
450      * a text title positioned on the left or right of a chart will be rotated
451      * 90 degrees.
452      *
453      * @param g2  the graphics device.
454      * @param widthRange  the width range.
455      * @param heightRange  the height range.
456      *
457      * @return The content size.
458      */
459     protected Size2D arrangeRR(Graphics2D g2, Range widthRange,
460             Range heightRange) {
461 
462         RectangleEdge position = getPosition();
463         if (position == RectangleEdge.TOP || position == RectangleEdge.BOTTOM) {
464 
465 
466             float maxWidth = (float) widthRange.getUpperBound();
467 
468             // determine the space required for the axis
469             AxisSpace space = this.axis.reserveSpace(g2, null,
470                     new Rectangle2D.Double(0, 0, maxWidth, 100),
471                     RectangleEdge.BOTTOM, null);
472 
473             return new Size2D(maxWidth, this.stripWidth + this.axisOffset
474                     + space.getTop() + space.getBottom());
475         }
476         else if (position == RectangleEdge.LEFT || position
477                 == RectangleEdge.RIGHT) {
478             float maxHeight = (float) heightRange.getUpperBound();
479             AxisSpace space = this.axis.reserveSpace(g2, null,
480                     new Rectangle2D.Double(0, 0, 100, maxHeight),
481                     RectangleEdge.RIGHT, null);
482             return new Size2D(this.stripWidth + this.axisOffset
483                     + space.getLeft() + space.getRight(), maxHeight);
484         }
485         else {
486             throw new RuntimeException("Unrecognised position.");
487         }
488     }

503      *
504      * @param g2  the graphics target ({@code null} not permitted).
505      * @param area  the drawing area ({@code null} not permitted).
506      * @param params  drawing parameters (ignored here).
507      *
508      * @return {@code null}.
509      */
510     @Override
511     public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
512         Rectangle2D target = (Rectangle2D) area.clone();
513         target = trimMargin(target);
514         if (this.backgroundPaint != null) {
515             g2.setPaint(this.backgroundPaint);
516             g2.fill(target);
517         }
518         getFrame().draw(g2, target);
519         getFrame().getInsets().trim(target);
520         target = trimPadding(target);
521         double base = this.axis.getLowerBound();
522         double increment = this.axis.getRange().getLength() / this.subdivisions;
523         Rectangle2D r = new Rectangle2D.Double();
524 
525         if (RectangleEdge.isTopOrBottom(getPosition())) {
526             RectangleEdge axisEdge = Plot.resolveRangeAxisLocation(
527                     this.axisLocation, PlotOrientation.HORIZONTAL);
528             if (axisEdge == RectangleEdge.TOP) {
529                 for (int i = 0; i < this.subdivisions; i++) {
530                     double v = base + (i * increment);
531                     Paint p = this.scale.getPaint(v);
532                     double vv0 = this.axis.valueToJava2D(v, target,
533                             RectangleEdge.TOP);
534                     double vv1 = this.axis.valueToJava2D(v + increment, target,
535                             RectangleEdge.TOP);
536                     double ww = Math.abs(vv1 - vv0) + 1.0;
537                     r.setRect(Math.min(vv0, vv1), target.getMaxY()
538                             - this.stripWidth, ww, this.stripWidth);
539                     g2.setPaint(p);
540                     g2.fill(r);
541                 }
542                 if (isStripOutlineVisible()) {
543                     g2.setPaint(this.stripOutlinePaint);
544                     g2.setStroke(this.stripOutlineStroke);
545                     g2.draw(new Rectangle2D.Double(target.getMinX(),
546                             target.getMaxY() - this.stripWidth,
547                             target.getWidth(), this.stripWidth));
548                 }
549                 this.axis.draw(g2, target.getMaxY() - this.stripWidth
550                         - this.axisOffset, target, target, RectangleEdge.TOP,
551                         null);
552             }
553             else if (axisEdge == RectangleEdge.BOTTOM) {
554                 for (int i = 0; i < this.subdivisions; i++) {
555                     double v = base + (i * increment);
556                     Paint p = this.scale.getPaint(v);
557                     double vv0 = this.axis.valueToJava2D(v, target,
558                             RectangleEdge.BOTTOM);
559                     double vv1 = this.axis.valueToJava2D(v + increment, target,
560                             RectangleEdge.BOTTOM);
561                     double ww = Math.abs(vv1 - vv0) + 1.0;
562                     r.setRect(Math.min(vv0, vv1), target.getMinY(), ww,
563                             this.stripWidth);
564                     g2.setPaint(p);
565                     g2.fill(r);
566                 }
567                 if (isStripOutlineVisible()) {
568                     g2.setPaint(this.stripOutlinePaint);
569                     g2.setStroke(this.stripOutlineStroke);
570                     g2.draw(new Rectangle2D.Double(target.getMinX(),
571                             target.getMinY(), target.getWidth(),
572                             this.stripWidth));
573                 }
574                 this.axis.draw(g2, target.getMinY() + this.stripWidth
575                         + this.axisOffset, target, target,
576                         RectangleEdge.BOTTOM, null);
577             }
578         }
579         else {
580             RectangleEdge axisEdge = Plot.resolveRangeAxisLocation(
581                     this.axisLocation, PlotOrientation.VERTICAL);
582             if (axisEdge == RectangleEdge.LEFT) {
583                 for (int i = 0; i < this.subdivisions; i++) {
584                     double v = base + (i * increment);
585                     Paint p = this.scale.getPaint(v);
586                     double vv0 = this.axis.valueToJava2D(v, target,
587                             RectangleEdge.LEFT);
588                     double vv1 = this.axis.valueToJava2D(v + increment, target,
589                             RectangleEdge.LEFT);
590                     double hh = Math.abs(vv1 - vv0) + 1.0;
591                     r.setRect(target.getMaxX() - this.stripWidth,
592                             Math.min(vv0, vv1), this.stripWidth, hh);
593                     g2.setPaint(p);
594                     g2.fill(r);
595                 }
596                 if (isStripOutlineVisible()) {
597                     g2.setPaint(this.stripOutlinePaint);
598                     g2.setStroke(this.stripOutlineStroke);
599                     g2.draw(new Rectangle2D.Double(target.getMaxX()
600                             - this.stripWidth, target.getMinY(), 
601                             this.stripWidth, target.getHeight()));
602                 }
603                 this.axis.draw(g2, target.getMaxX() - this.stripWidth
604                         - this.axisOffset, target, target, RectangleEdge.LEFT,
605                         null);
606             }
607             else if (axisEdge == RectangleEdge.RIGHT) {
608                 for (int i = 0; i < this.subdivisions; i++) {
609                     double v = base + (i * increment);
610                     Paint p = this.scale.getPaint(v);
611                     double vv0 = this.axis.valueToJava2D(v, target,
612                             RectangleEdge.LEFT);
613                     double vv1 = this.axis.valueToJava2D(v + increment, target,
614                             RectangleEdge.LEFT);
615                     double hh = Math.abs(vv1 - vv0) + 1.0;
616                     r.setRect(target.getMinX(), Math.min(vv0, vv1),
617                             this.stripWidth, hh);
618                     g2.setPaint(p);
619                     g2.fill(r);
620                 }
621                 if (isStripOutlineVisible()) {
622                     g2.setPaint(this.stripOutlinePaint);
623                     g2.setStroke(this.stripOutlineStroke);
624                     g2.draw(new Rectangle2D.Double(target.getMinX(),
