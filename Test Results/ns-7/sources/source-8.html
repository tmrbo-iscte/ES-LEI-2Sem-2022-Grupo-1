59      * Tests two shapes for equality.  If both shapes are {@code null},
60      * this method will return {@code true}.
61      * <p>
62      * In the current implementation, the following shapes are supported:
63      * {@code Ellipse2D}, {@code Line2D} and {@code Rectangle2D}
64      * (implicit).
65      *
66      * @param s1  the first shape ({@code null} permitted).
67      * @param s2  the second shape ({@code null} permitted).
68      *
69      * @return A boolean.
70      */
71     public static boolean equal(Shape s1, Shape s2) {
72         if (s1 instanceof Line2D && s2 instanceof Line2D) {
73             return equal((Line2D) s1, (Line2D) s2);
74         } else if (s1 instanceof Polygon && s2 instanceof Polygon) {
75             return equal((Polygon) s1, (Polygon) s2);
76         } else if (s1 instanceof Path2D && s2 instanceof Path2D) {
77             return equal((Path2D) s1, (Path2D) s2);
78         } else {
79             // this will handle Arc2D, Ellipse2D, Rectangle2D...
80             return Objects.equals(s1, s2);
81         }
82     }

138      * Tests two {@code GeneralPath} instances for equality.  If both are 
139      * {@code null} this method returns {@code true}.
140      *
141      * @param p1  path 1 ({@code null} permitted).
142      * @param p2  path 2 ({@code null} permitted).
143      *
144      * @return A boolean.
145      */
146     public static boolean equal(Path2D p1, Path2D p2) {
147         if (p1 == null) {
148             return (p2 == null);
149         }
150         if (p2 == null) {
151             return false;
152         }
153         if (p1.getWindingRule() != p2.getWindingRule()) {
154             return false;
155         }
156         PathIterator iterator1 = p1.getPathIterator(null);
157         PathIterator iterator2 = p2.getPathIterator(null);
158         double[] d1 = new double[6];
159         double[] d2 = new double[6];
160         boolean done = iterator1.isDone() && iterator2.isDone();
161         while (!done) {
162             if (iterator1.isDone() != iterator2.isDone()) {
163                 return false;
164             }
165             int seg1 = iterator1.currentSegment(d1);
166             int seg2 = iterator2.currentSegment(d2);
167             if (seg1 != seg2) {
168                 return false;
169             }
170             if (!Arrays.equals(d1, d2)) {
171                 return false;
172             }
173             iterator1.next();
174             iterator2.next();
175             done = iterator1.isDone() && iterator2.isDone();
176         }
177         return true;
178     }

270         final GeneralPath p0 = new GeneralPath();
271         p0.moveTo(-l - t, -l + t);
272         p0.lineTo(-l + t, -l - t);

278         p0.lineTo(l - t, l + t);
279         p0.lineTo(0.0f, t * SQRT2);

282         p0.lineTo(-t * SQRT2, 0.0f);

363      * Creates a region surrounding a line segment by 'widening' the line
364      * segment.  A typical use for this method is the creation of a
365      * 'clickable' region for a line that is displayed on-screen.
366      *
367      * @param line  the line ({@code null} not permitted).
368      * @param width  the width of the region.
369      *
370      * @return A region that surrounds the line.
371      */
372     public static Shape createLineRegion(Line2D line, float width) {
373         Args.nullNotPermitted(line, "line");
374         final GeneralPath result = new GeneralPath();
375         final float x1 = (float) line.getX1();
376         final float x2 = (float) line.getX2();
377         final float y1 = (float) line.getY1();
378         final float y2 = (float) line.getY2();
379         if ((x2 - x1) != 0.0) {
380             final double theta = Math.atan((y2 - y1) / (x2 - x1));
381             final float dx = (float) Math.sin(theta) * width;
382             final float dy = (float) Math.cos(theta) * width;
383             result.moveTo(x1 - dx, y1 + dy);
384             result.lineTo(x1 + dx, y1 - dy);
385             result.lineTo(x2 + dx, y2 - dy);
386             result.lineTo(x2 - dx, y2 + dy);
387             result.closePath();
388         } else {
389             // special case, vertical line
390             result.moveTo(x1 - width / 2.0f, y1);
391             result.lineTo(x1 + width / 2.0f, y1);
392             result.lineTo(x2 + width / 2.0f, y2);
393             result.lineTo(x2 - width / 2.0f, y2);
394             result.closePath();
395         }
396         return result;
397     }

495      * Returns {@code true} if the two maps contain the same set of entries and 
496      * {@code false} otherwise.
497      * 
498      * @param <K>  the key type.
499      * @param map1  the first map.
500      * @param map2  the second map.
501      * 
502      * @return A boolean.
503      */
504     public static <K extends Comparable<K>> boolean equal(Map<K, Shape> map1, Map<K, Shape> map2) {
505         if (!map1.keySet().equals(map2.keySet())) {
506             return false;
507         }
508         for (K key : map1.keySet()) {
509             Shape p1 = map1.get(key);
510             Shape p2 = map2.get(key);
511             if (!ShapeUtils.equal(p1, p2)) {
512                 return false;
513             }
514         }
515         return true;
516     }
