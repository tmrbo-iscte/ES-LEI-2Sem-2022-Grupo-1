46 /**
47  * Interface for a label generator for plots that use data from
48  * a {@link PieDataset}.
49  */
50 public interface PieSectionLabelGenerator<K extends Comparable<K>> {
51 
52     /**
53      * Generates a label for a pie section.
54      *
55      * @param dataset  the dataset ({@code null} not permitted).
56      * @param key  the section key ({@code null} not permitted).
57      *
58      * @return The label (possibly {@code null}).
59      */
60     String generateSectionLabel(PieDataset<K> dataset, K key);
61 
62     /**
63      * Generates an attributed label for the specified series, or
64      * {@code null} if no attributed label is available (in which case,
65      * the string returned by
66      * {@link #generateSectionLabel(PieDataset, Comparable)} will
67      * provide the fallback).  Only certain attributes are recognised by the
68      * code that ultimately displays the labels:
69      * <ul>
70      * <li>{@link TextAttribute#FONT}: will set the font;</li>
71      * <li>{@link TextAttribute#POSTURE}: a value of
72      *     {@link TextAttribute#POSTURE_OBLIQUE} will add {@link Font#ITALIC} to
73      *     the current font;</li>
74      * <li>{@link TextAttribute#WEIGHT}: a value of
75      *     {@link TextAttribute#WEIGHT_BOLD} will add {@link Font#BOLD} to the
76      *     current font;</li>
77      * <li>{@link TextAttribute#FOREGROUND}: this will set the {@link Paint}
78      *     for the current</li>
79      * <li>{@link TextAttribute#SUPERSCRIPT}: the values
80      *     {@link TextAttribute#SUPERSCRIPT_SUB} and
81      *     {@link TextAttribute#SUPERSCRIPT_SUPER} are recognised.</li>
82      * </ul>
83      *
84      * @param dataset  the dataset.
85      * @param key  the key.
86      *
87      * @return An attributed label (possibly {@code null}).
88      */
89     AttributedString generateAttributedSectionLabel(PieDataset<K> dataset,
90             K key);
91 
92 }
