47 /**
48  * A label generator for plots that use data from an
49  * {@link IntervalCategoryDataset}.
50  */
51 public class IntervalCategoryItemLabelGenerator
52     extends StandardCategoryItemLabelGenerator
53     implements CategoryItemLabelGenerator, PublicCloneable, Cloneable,
54                Serializable {
55 
56     /** For serialization. */
57     private static final long serialVersionUID = 5056909225610630529L;
58 
59     /** The default format string. */
60     public static final String DEFAULT_LABEL_FORMAT_STRING
61         = "({0}, {1}) = {3} - {4}";
62 
63     /**
64      * Creates a new generator with a default number formatter.
65      */
66     public IntervalCategoryItemLabelGenerator() {
67         super(DEFAULT_LABEL_FORMAT_STRING, NumberFormat.getInstance());
68     }
69 
70     /**
71      * Creates a new generator with the specified number formatter.
72      *
73      * @param labelFormat  the label format string ({@code null} not
74      *                     permitted).
75      * @param formatter  the number formatter ({@code null} not permitted).
76      */
77     public IntervalCategoryItemLabelGenerator(String labelFormat,
78                                               NumberFormat formatter) {
79         super(labelFormat, formatter);
80     }
81 
82     /**
83      * Creates a new generator with the specified date formatter.
84      *
85      * @param labelFormat  the label format string ({@code null} not
86      *                     permitted).
87      * @param formatter  the date formatter ({@code null} not permitted).
88      */
89     public IntervalCategoryItemLabelGenerator(String labelFormat,
90                                               DateFormat formatter) {
91         super(labelFormat, formatter);
92     }
93 
94     /**
95      * Creates the array of items that can be passed to the
96      * {@code MessageFormat} class for creating labels.
97      *
98      * @param dataset  the dataset ({@code null} not permitted).
99      * @param row  the row index (zero-based).
100      * @param column  the column index (zero-based).
101      *
102      * @return The items (never {@code null}).
103      */
104     @Override
105     protected Object[] createItemArray(CategoryDataset dataset,
106                                        int row, int column) {
107         Object[] result = new Object[5];
108         result[0] = dataset.getRowKey(row).toString();
109         result[1] = dataset.getColumnKey(column).toString();
110         Number value = dataset.getValue(row, column);
111         if (getNumberFormat() != null) {
112             result[2] = getNumberFormat().format(value);
113         }
114         else if (getDateFormat() != null) {
115             result[2] = getDateFormat().format(value);
116         }
117 
118         if (dataset instanceof IntervalCategoryDataset) {
119             IntervalCategoryDataset icd = (IntervalCategoryDataset) dataset;
120             Number start = icd.getStartValue(row, column);
121             Number end = icd.getEndValue(row, column);
122             if (getNumberFormat() != null) {
123                 result[3] = getNumberFormat().format(start);
124                 result[4] = getNumberFormat().format(end);
125             }
126             else if (getDateFormat() != null) {
127                 result[3] = getDateFormat().format(start);
128                 result[4] = getDateFormat().format(end);
129             }
130         }
131         return result;
132     }

134 }
