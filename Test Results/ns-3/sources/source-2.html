39 import java.util.ArrayList;
40 import java.util.List;

44 /**
45  * A collection of axes that have been assigned to the TOP, BOTTOM, LEFT or
46  * RIGHT of a chart.  This class is used internally by JFreeChart, you won't
47  * normally need to use it yourself.
48  */
49 public class AxisCollection {
50 
51     /** The axes that need to be drawn at the top of the plot area. */
52     private final List<Axis> axesAtTop;
53 
54     /** The axes that need to be drawn at the bottom of the plot area. */
55     private final List<Axis> axesAtBottom;
56 
57     /** The axes that need to be drawn at the left of the plot area. */
58     private final List<Axis> axesAtLeft;
59 
60     /** The axes that need to be drawn at the right of the plot area. */
61     private final List<Axis> axesAtRight;
62 
63     /**
64      * Creates a new empty collection.
65      */
66     public AxisCollection() {
67         this.axesAtTop = new ArrayList<>();
68         this.axesAtBottom = new ArrayList<>();
69         this.axesAtLeft = new ArrayList<>();
70         this.axesAtRight = new ArrayList<>();
71     }
72 
73     /**
74      * Returns a list of the axes (if any) that need to be drawn at the top of
75      * the plot area.
76      *
77      * @return A list of axes.
78      */
79     public List<Axis> getAxesAtTop() {
80         return this.axesAtTop;
81     }
82 
83    /**
84     * Returns a list of the axes (if any) that need to be drawn at the bottom
85     * of the plot area.
86     *
87     * @return A list of axes.
88     */
89    public List<Axis> getAxesAtBottom() {
90         return this.axesAtBottom;
91     }
92 
93     /**
94      * Returns a list of the axes (if any) that need to be drawn at the left
95      * of the plot area.
96      *
97      * @return A list of axes.
98      */
99     public List<Axis> getAxesAtLeft() {
100         return this.axesAtLeft;
101     }
102 
103     /**
104     * Returns a list of the axes (if any) that need to be drawn at the right
105     * of the plot area.
106     *
107     * @return A list of axes.
108     */
109     public List<Axis> getAxesAtRight() {
110         return this.axesAtRight;
111     }
112 
113     /**
114      * Adds an axis to the collection.
115      *
116      * @param axis  the axis ({@code null} not permitted).
117      * @param edge  the edge of the plot that the axis should be drawn on
118      *              ({@code null} not permitted).
119      */
120     public void add(Axis axis, RectangleEdge edge) {
121         Args.nullNotPermitted(axis, "axis");
122         Args.nullNotPermitted(edge, "edge");
123         switch (edge) {
124             case TOP:
125                 this.axesAtTop.add(axis);
126                 break;
127             case BOTTOM:
128                 this.axesAtBottom.add(axis);
129                 break;
130             case LEFT:
131                 this.axesAtLeft.add(axis);
132                 break;
133             case RIGHT:
134                 this.axesAtRight.add(axis);
135                 break;
136             default:
137                 break;
138         }
139     }
