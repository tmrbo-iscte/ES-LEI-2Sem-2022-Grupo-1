301     /**
302      * Draws the block within the specified area.
303      *
304      * @param g2  the graphics device.
305      * @param area  the area.
306      * @param params  ignored ({@code null} permitted).
307      *
308      * @return Always {@code null}.
309      */
310     @Override
311     public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
312         area = trimMargin(area);
313         drawBorder(g2, area);
314         area = trimBorder(area);
315         area = trimPadding(area);
316 
317         // check if we need to collect chart entities from the container
318         EntityBlockParams ebp = null;
319         StandardEntityCollection sec = null;
320         Shape entityArea = null;
321         if (params instanceof EntityBlockParams) {
322             ebp = (EntityBlockParams) params;
323             if (ebp.getGenerateEntities()) {
324                 sec = new StandardEntityCollection();
325                 entityArea = (Shape) area.clone();
326             }
327         }
328         g2.setPaint(this.paint);
329         g2.setFont(this.font);
330         Point2D pt = this.textAnchor.getAnchorPoint(area);
331         this.label.draw(g2, (float) pt.getX(), (float) pt.getY(),
332                 this.contentAlignmentPoint);
333         BlockResult result = null;
334         if (ebp != null && sec != null) {
335             if (this.toolTipText != null || this.urlText != null) {
336                 ChartEntity entity = new ChartEntity(entityArea,
337                         this.toolTipText, this.urlText);
338                 sec.add(entity);
339                 result = new BlockResult();
340                 result.setEntityCollection(sec);
341             }
342         }
343         return result;
344     }
