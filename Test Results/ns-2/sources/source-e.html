40 /**
41  * An interface that exposes the clone() method.  In order to support the 
42  * cloning of {@link JFreeChart} instances, it is advisable to implement this
43  * interface for custom plots, renderers and other chart components.  If
44  * this interface is not implemented, cloning will still be attempted via 
45  * reflection.
46  */

50      * Returns a clone of the object.
51      * 
52      * @return A clone.
53      *

55      */
56     Object clone() throws CloneNotSupportedException;
57 
58 }
