31 import java.awt.geom.Rectangle2D;
32 
33 /**
34  * Used to indicate the edge of a rectangle.
35  */
36 public enum RectangleEdge {
37 
38     /** Top. */
39     TOP,
40 
41     /** Bottom. */
42     BOTTOM,
43 
44     /** Left. */
45     LEFT,
46 
47     /** Right. */
48     RIGHT;
49 
50     /**
51      * Returns {@code true} if the edge is {@code TOP} or 
52      * {@code BOTTOM}, and {@code false} otherwise.
53      * 
54      * @param edge  the edge.
55      * 
56      * @return A boolean.
57      */
58     public static boolean isTopOrBottom(RectangleEdge edge) {
59         return (edge == RectangleEdge.TOP || edge == RectangleEdge.BOTTOM);    
60     }
61     
62     /**
63      * Returns {@code true} if the edge is {@code LEFT} or 
64      * {@code RIGHT}, and {@code false} otherwise.
65      * 
66      * @param edge  the edge.
67      * 
68      * @return A boolean.
69      */
70     public static boolean isLeftOrRight(RectangleEdge edge) {
71         return (edge == RectangleEdge.LEFT || edge == RectangleEdge.RIGHT);    
72     }
73 
74     /**
75      * Returns the opposite edge.
76      * 
77      * @param edge  an edge.
78      * 
79      * @return The opposite edge.
80      */
81     public static RectangleEdge opposite(RectangleEdge edge) {
82         switch (edge) {
83             case TOP:
84                 return RectangleEdge.BOTTOM;
85             case BOTTOM:
86                 return RectangleEdge.TOP;
87             case LEFT:
88                 return RectangleEdge.RIGHT;
89             case RIGHT:
90                 return RectangleEdge.LEFT;
91             default:
92                 return null;
93         }
94     }
95     
96     /**
97      * Returns the x or y coordinate of the specified edge.
98      * 
99      * @param rectangle  the rectangle.
100      * @param edge  the edge.
101      * 
102      * @return The coordinate.
103      */
104     public static double coordinate(Rectangle2D rectangle, RectangleEdge edge) {
105         switch (edge) {
106             case TOP:
107                 return rectangle.getMinY();
108             case BOTTOM:
109                 return rectangle.getMaxY();
110             case LEFT:
111                 return rectangle.getMinX();
112             case RIGHT:
113                 return rectangle.getMaxX();
114             default:
115                 return 0.0;
116         }
117     }

119 }
