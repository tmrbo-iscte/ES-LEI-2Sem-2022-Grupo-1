46 /**
47  * A tool tip generator that stores custom tooltips. The dataset passed into
48  * the generateToolTip method is ignored.

50 public class CustomXYToolTipGenerator implements XYToolTipGenerator,

56     /** Storage for the tooltip lists. */

58 
59     /**
60      * Default constructor.
61      */
62     public CustomXYToolTipGenerator() {

64     }

66     /**
67      * Returns the number of tool tip lists stored by the renderer.
68      *
69      * @return The list count.
70      */
71     public int getListCount() {
72         return this.toolTipSeries.size();
73     }
74 
75     /**
76      * Returns the number of tool tips in a given list.
77      *
78      * @param list  the list index (zero based).
79      *
80      * @return The tooltip count.
81      */
82     public int getToolTipCount(int list) {
83 
84         int result = 0;
85         List<String> tooltips = this.toolTipSeries.get(list);
86         if (tooltips != null) {
87             result = tooltips.size();
88         }
89         return result;
90     }
91 
92     /**
93      * Returns the tool tip text for an item.
94      *
95      * @param series  the series index.
96      * @param item  the item index.
97      *
98      * @return The tool tip text (possibly {@code null}).
99      */
100     public String getToolTipText(int series, int item) {
101         String result = null;
102         if (series < getListCount()) {
103             List<String> tooltips = this.toolTipSeries.get(series);
104             if (tooltips != null) {
105                 if (item < tooltips.size()) {
106                     result = tooltips.get(item);
107                 }
108             }
109         }
110         return result;
111     }
112 
113     /**
114      * Adds a list of tooltips for a series.
115      *
116      * @param toolTips  the list of tool tips.
117      */
118     public void addToolTipSeries(List<String> toolTips) {
119         this.toolTipSeries.add(toolTips);
120     }
121 
122     /**

124      *
125      * @param data  the dataset (ignored in this implementation).
126      * @param series  the series (zero-based index).
127      * @param item  the item (zero-based index).
128      *
129      * @return The tooltip text.
130      */
131     @Override
132     public String generateToolTip(XYDataset data, int series, int item) {

134     }
