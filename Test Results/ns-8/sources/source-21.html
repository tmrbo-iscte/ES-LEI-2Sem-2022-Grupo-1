41 /**
42  * A <i>category item label generator</i> is an object that can be assigned to a
43  * {@link org.jfree.chart.renderer.category.CategoryItemRenderer} and that
44  * assumes responsibility for creating text items to be used as labels for the
45  * items in a {@link org.jfree.chart.plot.CategoryPlot}.
46  * <p>
47  * To assist with cloning charts, classes that implement this interface should
48  * also implement the {@link org.jfree.chart.api.PublicCloneable} interface.
49  */
50 public interface CategoryItemLabelGenerator {
51 
52     /**
53      * Generates a label for the specified row.
54      *
55      * @param dataset  the dataset ({@code null} not permitted).
56      * @param row  the row index (zero-based).
57      *
58      * @return The label.
59      */
60     String generateRowLabel(CategoryDataset dataset, int row);
61 
62     /**
63      * Generates a label for the specified row.
64      *
65      * @param dataset  the dataset ({@code null} not permitted).
66      * @param column  the column index (zero-based).
67      *
68      * @return The label.
69      */
70     String generateColumnLabel(CategoryDataset dataset, int column);
71 
72     /**
73      * Generates a label for the specified item. The label is typically a
74      * formatted version of the data value, but any text can be used.
75      *
76      * @param dataset  the dataset ({@code null} not permitted).
77      * @param row  the row index (zero-based).
78      * @param column  the column index (zero-based).
79      *
80      * @return The label (possibly {@code null}).
81      */
82     String generateLabel(CategoryDataset dataset, int row, int column);
83 
84 }
