389     /**
390      * Fetches the latest legend items.
391      */
392     protected void fetchLegendItems() {
393         this.items.clear();
394         RectangleEdge p = getPosition();
395         if (RectangleEdge.isTopOrBottom(p)) {
396             this.items.setArrangement(this.hLayout);
397         }
398         else {
399             this.items.setArrangement(this.vLayout);
400         }
401 
402         if (this.sortOrder.equals(SortOrder.ASCENDING)) {
403             for (int s = 0; s < this.sources.length; s++) {
404                 LegendItemCollection legendItems =
405                     this.sources[s].getLegendItems();
406                 if (legendItems != null) {
407                     for (int i = 0; i < legendItems.getItemCount(); i++) {
408                         addItemBlock(legendItems.get(i));
409                     }
410                 }
411             }
412         }
413         else {
414             for (int s = this.sources.length - 1; s >= 0; s--) {
415                 LegendItemCollection legendItems =
416                     this.sources[s].getLegendItems();
417                 if (legendItems != null) {
418                     for (int i = legendItems.getItemCount()-1; i >= 0; i--) {
419                         addItemBlock(legendItems.get(i));
420                     }
421                 }
422             }
423         }
424     }

432      * Creates a legend item block.
433      *
434      * @param item  the legend item.
435      *
436      * @return The block.
437      */
438     protected Block createLegendItemBlock(LegendItem item) {
439         BlockContainer result;
440         LegendGraphic lg = new LegendGraphic(item.getShape(),
441                 item.getFillPaint());
442         lg.setFillPaintTransformer(item.getFillPaintTransformer());
443         lg.setShapeFilled(item.isShapeFilled());
444         lg.setLine(item.getLine());
445         lg.setLineStroke(item.getLineStroke());
446         lg.setLinePaint(item.getLinePaint());
447         lg.setLineVisible(item.isLineVisible());
448         lg.setShapeVisible(item.isShapeVisible());
449         lg.setShapeOutlineVisible(item.isShapeOutlineVisible());
450         lg.setOutlinePaint(item.getOutlinePaint());
451         lg.setOutlineStroke(item.getOutlineStroke());
452         lg.setPadding(this.legendItemGraphicPadding);
453 
454         LegendItemBlockContainer legendItem = new LegendItemBlockContainer(
455                 new BorderArrangement(), item.getDataset(),
456                 item.getSeriesKey());
457         lg.setShapeAnchor(getLegendItemGraphicAnchor());
458         lg.setShapeLocation(getLegendItemGraphicLocation());
459         legendItem.add(lg, this.legendItemGraphicEdge);
460         Font textFont = item.getLabelFont();
461         if (textFont == null) {
462             textFont = this.itemFont;
463         }
464         Paint textPaint = item.getLabelPaint();
465         if (textPaint == null) {
466             textPaint = this.itemPaint;
467         }
468         LabelBlock labelBlock = new LabelBlock(item.getLabel(), textFont,
469                 textPaint);
470         labelBlock.setPadding(this.itemLabelPadding);
471         legendItem.add(labelBlock);
472         legendItem.setToolTipText(item.getToolTipText());
473         legendItem.setURLText(item.getURLText());
474 
475         result = new BlockContainer(new CenterArrangement());
476         result.add(legendItem);
477 
478         return result;
479     }

502         fetchLegendItems();
503         if (this.items.isEmpty()) {

512         result.height = calculateTotalHeight(size.height);
513         result.width = calculateTotalWidth(size.width);
514         return result;
515     }

540     /**
541      * Draws the block within the specified area.
542      *
543      * @param g2  the graphics device.
544      * @param area  the area.
545      * @param params  ignored ({@code null} permitted).
546      *
547      * @return An {@link org.jfree.chart.block.EntityBlockResult} or
548      *         {@code null}.
549      */
550     @Override
551     public Object draw(Graphics2D g2, Rectangle2D area, Object params) {
552         Rectangle2D target = (Rectangle2D) area.clone();
553         Rectangle2D hotspot = (Rectangle2D) area.clone();
554         StandardEntityCollection sec = null;
555         if (params instanceof EntityBlockParams
556                 && ((EntityBlockParams) params).getGenerateEntities()) {
557             sec = new StandardEntityCollection();
558             sec.add(new TitleEntity(hotspot, this));
559         }
560         target = trimMargin(target);
561         if (this.backgroundPaint != null) {
562             g2.setPaint(this.backgroundPaint);
563             g2.fill(target);
564         }
565         BlockFrame border = getFrame();
566         border.draw(g2, target);
567         border.getInsets().trim(target);
568         BlockContainer container = this.wrapper;
569         if (container == null) {
570             container = this.items;
571         }
572         target = trimPadding(target);
573         Object val = container.draw(g2, target, params);
574         if (val instanceof BlockResult) {
575             EntityCollection ec = ((BlockResult) val).getEntityCollection();
576             if (ec != null && sec != null) {
577                 sec.addAll(ec);
578                 ((BlockResult) val).setEntityCollection(sec);
579             }
580         }
581         return val;
582     }
