301     /**
302      * Returns the new size that reflects the constraints defined by this
303      * instance.
304      *
305      * @param base  the base size.
306      *
307      * @return The constrained size.
308      */
309     public Size2D calculateConstrainedSize(Size2D base) {
310         Size2D result = new Size2D();
311         if (this.widthConstraintType == LengthConstraintType.NONE) {
312             result.width = base.width;
313             if (this.heightConstraintType == LengthConstraintType.NONE) {
314                result.height = base.height;
315             }
316             else if (this.heightConstraintType == LengthConstraintType.RANGE) {
317                result.height = this.heightRange.constrain(base.height);
318             }
319             else if (this.heightConstraintType == LengthConstraintType.FIXED) {
320                result.height = this.height;
321             }
322         }
323         else if (this.widthConstraintType == LengthConstraintType.RANGE) {
324             result.width = this.widthRange.constrain(base.width);
325             if (this.heightConstraintType == LengthConstraintType.NONE) {
326                 result.height = base.height;
327             }
328             else if (this.heightConstraintType == LengthConstraintType.RANGE) {
329                 result.height = this.heightRange.constrain(base.height);
330             }
331             else if (this.heightConstraintType == LengthConstraintType.FIXED) {
332                 result.height = this.height;
333             }
334         }
335         else if (this.widthConstraintType == LengthConstraintType.FIXED) {
336             result.width = this.width;
337             if (this.heightConstraintType == LengthConstraintType.NONE) {
338                 result.height = base.height;
339             }
340             else if (this.heightConstraintType == LengthConstraintType.RANGE) {
341                 result.height = this.heightRange.constrain(base.height);
342             }
343             else if (this.heightConstraintType == LengthConstraintType.FIXED) {
344                 result.height = this.height;
345             }
346         }
347         return result;
348     }
