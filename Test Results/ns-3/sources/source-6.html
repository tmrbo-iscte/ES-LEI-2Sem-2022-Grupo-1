44 /**
45  * Instances of this class are used to carry state information for an axis
46  * during the drawing process.  By retaining this information in a separate
47  * object, it is possible for multiple threads to draw the same axis to
48  * different output targets (each drawing will maintain separate state
49  * information).
50  */
51 public class AxisState {
52 
53     /** The cursor position. */
54     private double cursor;
55 
56     /** The axis ticks. */
57     private List<ValueTick> ticks;
58 
59     /** The maximum width/height. */
60     private double max;
61 
62     /**
63      * Creates a new axis state.
64      */
65     public AxisState() {
66         this(0.0);
67     }
68 
69     /**
70      * Creates a new axis state.
71      *
72      * @param cursor  the cursor.
73      */
74     public AxisState(double cursor) {
75         this.cursor = cursor;
76         this.ticks = new ArrayList<>();
77     }
78 
79     /**
80      * Returns the cursor position.
81      *
82      * @return The cursor position.
83      */
84     public double getCursor() {
85         return this.cursor;
86     }
87 
88     /**
89      * Sets the cursor position.
90      *
91      * @param cursor  the cursor position.
92      */
93     public void setCursor(double cursor) {
94         this.cursor = cursor;
95     }
96 
97     /**
98      * Moves the cursor outwards by the specified number of units.
99      *
100      * @param units  the units.
101      * @param edge  the edge ({@code null} not permitted).
102      */
103     public void moveCursor(double units, RectangleEdge edge) {
104         Args.nullNotPermitted(edge, "edge");
105         switch (edge) {
106             case TOP:
107                 cursorUp(units);
108                 break;
109             case BOTTOM:
110                 cursorDown(units);
111                 break;
112             case LEFT:
113                 cursorLeft(units);
114                 break;
115             case RIGHT:
116                 cursorRight(units);
117                 break;
118             default:
119                 throw new IllegalStateException("Unexpected enum value " + edge);
120         }
121     }
122 
123     /**
124      * Moves the cursor up by the specified number of Java 2D units.
125      *
126      * @param units  the units.
127      */
128     public void cursorUp(double units) {
129         this.cursor = this.cursor - units;
130     }
131 
132     /**

136      */
137     public void cursorDown(double units) {

139     }
140 
141     /**

145      */
146     public void cursorLeft(double units) {
147         this.cursor = this.cursor - units;
148     }
149 
150     /**

152      *
153      * @param units  the units.
154      */
155     public void cursorRight(double units) {
156         this.cursor = this.cursor + units;
157     }
158
