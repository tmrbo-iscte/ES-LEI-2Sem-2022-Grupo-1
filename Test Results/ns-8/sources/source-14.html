39 import java.io.Serializable;
40 import java.util.Formatter;
41 import java.util.Objects;
42 import org.jfree.chart.internal.Args;
43 import org.jfree.data.flow.FlowDataset;

46 /**
47  * Standard flow label generator.  Instances of this class are immutable.
48  * 
49  * @since 1.5.3
50  */
51 public class StandardFlowLabelGenerator implements FlowLabelGenerator, Serializable {
52     
53     /** The default template for formatting the label. */
54     public static final String DEFAULT_TEMPLATE = "%2$s to %3$s = %4$,.2f";
55     
56     /** The template. */
57     private String template;
58     
59     /**
60      * Creates a new instance with the default template.
61      */
62     public StandardFlowLabelGenerator() {
63         this(DEFAULT_TEMPLATE);    
64     }
65     
66     /**
67      * Creates a new generator with the specified template.  The template
68      * is passed to a Java Formatter instance along with four arguments, the
69      * stage (an integer), the source (a String), the destination (a String)
70      * and the flow value (a Number).
71      * 
72      * @param template  the template ({@code null} not permitted). 
73      */
74     public StandardFlowLabelGenerator(String template) {
75         Args.nullNotPermitted(template, "template");
76         this.template = template;
77     }
78 
79     /**
80      * Returns a label for the specified flow.
81      * 
82      * @param dataset  the flow dataset ({@code null} not permitted).
83      * @param key  the flow key ({@code null} not permitted).
84      * 
85      * @return The label (possibly {@code null}). 
86      */
87     @Override
88     public String generateLabel(FlowDataset dataset, FlowKey key) {
89         Args.nullNotPermitted(dataset, "dataset");
90         Args.nullNotPermitted(key, "key");
91         String result;
92         try (Formatter formatter = new Formatter(new StringBuilder())) {
93             Number value = dataset.getFlow(key.getStage(), key.getSource(), key.getDestination());
94             formatter.format(this.template, key.getStage(), key.getSource(), key.getDestination(), value);
95             result = formatter.toString();
96         }
97         return result;
98     }
99     
100     /**
101      * Tests this instance for equality with an arbitrary object.
102      * 
103      * @param obj  the object to test ({@code null} permitted).
104      * 
105      * @return A boolean. 
106      */
107     @Override
108     public boolean equals(Object obj) {
109         if (!(obj instanceof StandardFlowLabelGenerator)) {
110             return false;
111         }
112         StandardFlowLabelGenerator that = (StandardFlowLabelGenerator) obj;
113         if (!this.template.equals(that.template)) {
114             return false;
115         }
116         return true;
117     }
118 
119     @Override
120     public int hashCode() {
121         int hash = 3;
122         hash = 97 * hash + Objects.hashCode(this.template);
123         return hash;
124     }

126 }
