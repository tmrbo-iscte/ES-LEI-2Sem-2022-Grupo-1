47 /**
48  * An item label generator for plots that use data from a
49  * {@link BoxAndWhiskerCategoryDataset}.
50  * <P>
51  * The tooltip text and item label text are composed using a
52  * {@link java.text.MessageFormat} object, that can aggregate some or all of
53  * the following string values into a message.
54  * <ul>
55  * <li>0 : Series Name</li>
56  * <li>1 : X (value or date)</li>
57  * <li>2 : Mean</li>
58  * <li>3 : Median</li>
59  * <li>4 : Minimum</li>
60  * <li>5 : Maximum</li>
61  * <li>6 : Quartile 1</li>
62  * <li>7 : Quartile 3</li>
63  * </ul>
64  */
65 public class BoxAndWhiskerToolTipGenerator
66         extends StandardCategoryToolTipGenerator
67         implements CategoryToolTipGenerator, Cloneable, PublicCloneable,
68                    Serializable {
69 
70     /** For serialization. */
71     private static final long serialVersionUID = -6076837753823076334L;
72 
73     /** The default tooltip format string. */
74     public static final String DEFAULT_TOOL_TIP_FORMAT
75             = "X: {1} Mean: {2} Median: {3} Min: {4} Max: {5} Q1: {6} Q3: {7} ";
76 
77     /**
78      * Creates a default tool tip generator.
79      */
80     public BoxAndWhiskerToolTipGenerator() {
81         super(DEFAULT_TOOL_TIP_FORMAT, NumberFormat.getInstance());
82     }
83 
84     /**
85      * Creates a tool tip formatter.
86      *
87      * @param format  the tool tip format string.
88      * @param formatter  the formatter.
89      */
90     public BoxAndWhiskerToolTipGenerator(String format,
91                                          NumberFormat formatter) {
92         super(format, formatter);
93     }
94 
95     /**
96      * Creates the array of items that can be passed to the
97      * {@link MessageFormat} class for creating labels.
98      *
99      * @param dataset  the dataset ({@code null} not permitted).
100      * @param series  the series (zero-based index).
101      * @param item  the item (zero-based index).
102      *
103      * @return The items (never {@code null}).
104      */
105     @Override
106     protected Object[] createItemArray(CategoryDataset dataset, int series,
107                                        int item) {
108         Object[] result = new Object[8];
109         result[0] = dataset.getRowKey(series);
110         Number y = dataset.getValue(series, item);
111         NumberFormat formatter = getNumberFormat();
112         result[1] = formatter.format(y);
113         if (dataset instanceof BoxAndWhiskerCategoryDataset) {
114             BoxAndWhiskerCategoryDataset d
115                 = (BoxAndWhiskerCategoryDataset) dataset;
116             result[2] = formatter.format(d.getMeanValue(series, item));
117             result[3] = formatter.format(d.getMedianValue(series, item));
118             result[4] = formatter.format(d.getMinRegularValue(series, item));
119             result[5] = formatter.format(d.getMaxRegularValue(series, item));
120             result[6] = formatter.format(d.getQ1Value(series, item));
121             result[7] = formatter.format(d.getQ3Value(series, item));
122         }
123         return result;
124     }
