47 /**
48  * A standard series label generator for plots that use data from
49  * an {@link org.jfree.data.xy.XYDataset}.
50  * <br><br>
51  * This class implements {@code PublicCloneable} by mistake but we retain
52  * this for the sake of backward compatibility.
53  */
54 public class StandardXYSeriesLabelGenerator implements XYSeriesLabelGenerator,
55         Cloneable, PublicCloneable, Serializable {
56 
57     /** For serialization. */
58     private static final long serialVersionUID = 1916017081848400024L;
59 
60     /** The default item label format. */
61     public static final String DEFAULT_LABEL_FORMAT = "{0}";
62 
63     /** The format pattern. */
64     private final String formatPattern;
65 
66     /**
67      * Creates a default series label generator (uses
68      * {@link #DEFAULT_LABEL_FORMAT}).
69      */
70     public StandardXYSeriesLabelGenerator() {
71         this(DEFAULT_LABEL_FORMAT);
72     }
73 
74     /**
75      * Creates a new series label generator.
76      *
77      * @param format  the format pattern ({@code null} not permitted).
78      */
79     public StandardXYSeriesLabelGenerator(String format) {
80         Args.nullNotPermitted(format, "format");
81         this.formatPattern = format;
82     }
83 
84     /**
85      * Generates a label for the specified series.  This label will be
86      * used for the chart legend.
87      *
88      * @param dataset  the dataset ({@code null} not permitted).
89      * @param series  the series.
90      *
91      * @return A series label.
92      */
93     @Override
94     public String generateLabel(XYDataset dataset, int series) {
95         Args.nullNotPermitted(dataset, "dataset");
96         String label = MessageFormat.format(
97             this.formatPattern, createItemArray(dataset, series)
98         );
99         return label;
100     }
101 
102     /**
103      * Creates the array of items that can be passed to the
104      * {@link MessageFormat} class for creating labels.
105      *
106      * @param dataset  the dataset ({@code null} not permitted).
107      * @param series  the series (zero-based index).
108      *
109      * @return The items (never {@code null}).
110      */
111     protected Object[] createItemArray(XYDataset dataset, int series) {
112         Object[] result = new Object[1];
113         result[0] = dataset.getSeriesKey(series).toString();
114         return result;
115     }
116

120      * behaviour for backwards compatibility.

123      *
124      * @throws CloneNotSupportedException if cloning is not supported.
125      */
126     @Override
127     public Object clone() throws CloneNotSupportedException {
128         return super.clone();
129     }
