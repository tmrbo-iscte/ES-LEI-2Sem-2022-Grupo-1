39 /**
40  * A utility class for checking method arguments.
41  */
42 public class Args {
43 
44     /**
45      * Throws an {@code IllegalArgumentException} if the supplied 
46      * {@code param} is {@code null}.
47      *
48      * @param param  the parameter to check ({@code null} permitted).
49      * @param name  the name of the parameter (to use in the exception message
50      *     if {@code param} is {@code null}).
51      *
52      * @throws IllegalArgumentException  if {@code param} is 
53      *     {@code null}.
54      */
55     public static void nullNotPermitted(Object param, String name) {
56         if (param == null) {
57             throw new IllegalArgumentException("Null '" + name + "' argument.");
58         }
59     }
60     
61     /**
62      * Throws an {@code IllegalArgumentException} if {@code value} is negative.
63      * 
64      * @param value  the value.
65      * @param name  the parameter name (for use in the exception message).
66      */
67     public static void requireNonNegative(int value, String name) {
68         if (value < 0) {
69             throw new IllegalArgumentException("Require '" + name + "' (" 
70                     + value + ") to be non-negative.");
71         }
72     }
73     
74     /**
75      * Throws an {@code IllegalArgumentException} if {@code value} is negative.
76      * 
77      * @param value  the value.
78      * @param name  the parameter name (for use in the exception message).
79      */
80     public static void requireNonNegative(double value, String name) {
81         if (value < 0) {
82             throw new IllegalArgumentException("Require '" + name + "' (" 
83                     + value + ") to be non-negative.");
84         }
85     }
86     
87     /**
88      * Checks that the value falls within the specified range and, if it does
89      * not, throws an {@code IllegalArgumentException}.
90      * 
91      * @param value  the value.
92      * @param name  the parameter name.
93      * @param lowerBound  the lower bound of the permitted range.
94      * @param upperBound  the upper bound fo the permitted range.
95      */
96     public static void requireInRange(int value, String name, 
97             int lowerBound, int upperBound) {
98         if (value < lowerBound || value > upperBound) {
99             throw new IllegalArgumentException("Require '" + name + "' (" 
100                     + value + ") to be in the range " + lowerBound + " to " 
101                     + upperBound);
102         }
103     }
104     
105     /**
106      * Checks the supplied value is finite (neither infinite nor NaN) and 
107      * throws an {@code IllegalArgumentException} if the requirement is not
108      * met.
109      * 
110      * @param value  the value.
111      * @param name  the parameter name (for use in the exception message).
112      * 
113      * @since 1.5.4
114      */
115     public static void requireFinite(double value, String name) {
116         if (!Double.isFinite(value)) {
117             throw new IllegalArgumentException("Require '" + name + "' (" 
118                     + value + ") to be finite.");
119         }
120     }
