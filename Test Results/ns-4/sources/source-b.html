105      * Calculates and sets the bounds of all the items in the specified
106      * container, subject to the given constraint.  The {@code Graphics2D}
107      * can be used by some items (particularly items containing text) to
108      * calculate sizing parameters.
109      *
110      * @param container  the container whose items are being arranged.
111      * @param constraint  the size constraint.
112      * @param g2  the graphics device.
113      *
114      * @return The size of the container after arrangement of the contents.
115      */
116     @Override
117     public Size2D arrange(BlockContainer container, Graphics2D g2,
118                           RectangleConstraint constraint) {
119 
120         LengthConstraintType w = constraint.getWidthConstraintType();
121         LengthConstraintType h = constraint.getHeightConstraintType();
122         if (w == LengthConstraintType.NONE) {
123             if (h == LengthConstraintType.NONE) {
124                 return arrangeNN(container, g2);
125             }
126             else if (h == LengthConstraintType.FIXED) {
127                 return arrangeNF(container, g2, constraint);
128             }
129             else if (h == LengthConstraintType.RANGE) {
130                 throw new RuntimeException("Not implemented.");
131             }
132         }
133         else if (w == LengthConstraintType.FIXED) {
134             if (h == LengthConstraintType.NONE) {
135                 return arrangeFN(container, g2, constraint);
136             }
137             else if (h == LengthConstraintType.FIXED) {
138                 return arrangeFF(container, g2, constraint);
139             }
140             else if (h == LengthConstraintType.RANGE) {
141                 return arrangeFR(container, g2, constraint);
142             }
143         }
144         else if (w == LengthConstraintType.RANGE) {
145             if (h == LengthConstraintType.NONE) {
146                 return arrangeRN(container, g2, constraint);
147             }
148             else if (h == LengthConstraintType.FIXED) {
149                 return arrangeRF(container, g2, constraint);
150             }
151             else if (h == LengthConstraintType.RANGE) {
152                 return arrangeRR(container, g2, constraint);
153             }
154         }
155         throw new RuntimeException("Unrecognised constraint type.");
156 
157     }

159     /**
160      * Arranges the blocks in the container with a fixed width and no height
161      * constraint.
162      *
163      * @param container  the container.
164      * @param constraint  the constraint.
165      * @param g2  the graphics device.
166      *
167      * @return The size.
168      */
169     protected Size2D arrangeFN(BlockContainer container, Graphics2D g2,
170                                RectangleConstraint constraint) {
171 
172         List<Block> blocks = container.getBlocks();
173         double width = constraint.getWidth();
174 
175         double x = 0.0;
176         double y = 0.0;
177         double maxHeight = 0.0;
178         List<Block> itemsInRow = new ArrayList<>();
179         for (Block block : blocks) {
180             Size2D size = block.arrange(g2, RectangleConstraint.NONE);
181             if (x + size.width <= width) {
182                 itemsInRow.add(block);
183                 block.setBounds(
184                     new Rectangle2D.Double(x, y, size.width, size.height)
185                 );
186                 x = x + size.width + this.horizontalGap;
187                 maxHeight = Math.max(maxHeight, size.height);
188             }
189             else {
190                 if (itemsInRow.isEmpty()) {
191                     // place in this row (truncated) anyway
192                     block.setBounds(
193                         new Rectangle2D.Double(
194                             x, y, Math.min(size.width, width - x), size.height
195                         )
196                     );
197                     x = 0.0;
198                     y = y + size.height + this.verticalGap;
199                 }
200                 else {
201                     // start new row
202                     itemsInRow.clear();
203                     x = 0.0;
204                     y = y + maxHeight + this.verticalGap;
205                     maxHeight = size.height;
206                     block.setBounds(
207                         new Rectangle2D.Double(
208                             x, y, Math.min(size.width, width), size.height
209                         )
210                     );
211                     x = size.width + this.horizontalGap;
212                     itemsInRow.add(block);
213                 }
214             }
215         }
216         return new Size2D(constraint.getWidth(), y + maxHeight);
217     }

348     protected Size2D arrangeNN(BlockContainer container, Graphics2D g2) {
349         double x = 0.0;
350         double width = 0.0;
351         double maxHeight = 0.0;
352         List<Block> blocks = container.getBlocks();
353         int blockCount = blocks.size();
354         if (blockCount > 0) {
355             Size2D[] sizes = new Size2D[blocks.size()];
356             for (int i = 0; i < blocks.size(); i++) {
357                 Block block = blocks.get(i);
358                 sizes[i] = block.arrange(g2, RectangleConstraint.NONE);
359                 width = width + sizes[i].getWidth();
360                 maxHeight = Math.max(sizes[i].height, maxHeight);
361                 block.setBounds(
362                     new Rectangle2D.Double(
363                         x, 0.0, sizes[i].width, sizes[i].height
364                     )
365                 );
366                 x = x + sizes[i].width + this.horizontalGap;
367             }
368             if (blockCount > 1) {
369                 width = width + this.horizontalGap * (blockCount - 1);
370             }
371             if (this.verticalAlignment != VerticalAlignment.TOP) {
372                 for (int i = 0; i < blocks.size(); i++) {
373                     //Block b = (Block) blocks.get(i);
374                     if (this.verticalAlignment == VerticalAlignment.CENTER) {

377                     else if (this.verticalAlignment
378                             == VerticalAlignment.BOTTOM) {
379                         //TODO: shift block down to bottom

381                 }
