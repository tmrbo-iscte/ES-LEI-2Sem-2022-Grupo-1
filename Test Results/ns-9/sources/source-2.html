451     /**
452      * Returns a string containing the characters from the given iterator.
453      *
454      * @param iterator  the iterator ({@code null} not permitted).
455      *
456      * @return A string.
457      */
458     private String characterIteratorToString(CharacterIterator iterator) {
459         int endIndex = iterator.getEndIndex();
460         int beginIndex = iterator.getBeginIndex();
461         int count = endIndex - beginIndex;
462         if (count <= 0) {
463             return "";
464         }
465         char[] chars = new char[count];
466         int i = 0;
467         char c = iterator.first();
468         while (c != CharacterIterator.DONE) {
469             chars[i] = c;
470             i++;
471             c = iterator.next();
472         }
473         return new String(chars);
474     }
