59     /**
60      * Returns {@code true} if the two {@code Paint} objects are equal 
61      * OR both {@code null}.  This method handles
62      * {@code GradientPaint}, {@code LinearGradientPaint} and 
63      * {@code RadialGradientPaint} as a special cases, since those classes do
64      * not override the {@code equals()} method.
65      *
66      * @param p1  paint 1 ({@code null} permitted).
67      * @param p2  paint 2 ({@code null} permitted).
68      *
69      * @return A boolean.
70      */
71     public static boolean equal(Paint p1, Paint p2) {
72         if (p1 == p2) {
73             return true;
74         }
75             
76         // handle cases where either or both arguments are null
77         if (p1 == null) {
78             return (p2 == null);   
79         }
80         if (p2 == null) {
81             return false;   
82         }
83 
84         // handle GradientPaint as a special case...
85         if (p1 instanceof GradientPaint && p2 instanceof GradientPaint) {
86             GradientPaint gp1 = (GradientPaint) p1;
87             GradientPaint gp2 = (GradientPaint) p2;
88             return gp1.getColor1().equals(gp2.getColor1()) 
89                     && gp1.getColor2().equals(gp2.getColor2())
90                     && gp1.getPoint1().equals(gp2.getPoint1())    
91                     && gp1.getPoint2().equals(gp2.getPoint2())
92                     && gp1.isCyclic() == gp2.isCyclic()
93                     && gp1.getTransparency() == gp1.getTransparency(); 
94         } else if (p1 instanceof LinearGradientPaint 
95                 && p2 instanceof LinearGradientPaint) {
96             LinearGradientPaint lgp1 = (LinearGradientPaint) p1;
97             LinearGradientPaint lgp2 = (LinearGradientPaint) p2;
98             return lgp1.getStartPoint().equals(lgp2.getStartPoint())
99                     && lgp1.getEndPoint().equals(lgp2.getEndPoint()) 
100                     && Arrays.equals(lgp1.getFractions(), lgp2.getFractions())
101                     && Arrays.equals(lgp1.getColors(), lgp2.getColors())
102                     && lgp1.getCycleMethod() == lgp2.getCycleMethod()
103                     && lgp1.getColorSpace() == lgp2.getColorSpace()
104                     && lgp1.getTransform().equals(lgp2.getTransform());
105         } else if (p1 instanceof RadialGradientPaint 
106                 && p2 instanceof RadialGradientPaint) {
107             RadialGradientPaint rgp1 = (RadialGradientPaint) p1;
108             RadialGradientPaint rgp2 = (RadialGradientPaint) p2;
109             return rgp1.getCenterPoint().equals(rgp2.getCenterPoint())
110                     && rgp1.getRadius() == rgp2.getRadius() 
111                     && rgp1.getFocusPoint().equals(rgp2.getFocusPoint())
112                     && Arrays.equals(rgp1.getFractions(), rgp2.getFractions())
113                     && Arrays.equals(rgp1.getColors(), rgp2.getColors())
114                     && rgp1.getCycleMethod() == rgp2.getCycleMethod()
115                     && rgp1.getColorSpace() == rgp2.getColorSpace()
116                     && rgp1.getTransform().equals(rgp2.getTransform());
117         } else {
118             return p1.equals(p2);
119         }
120     }

123      * Returns {@code true} if the two maps contain the same set of entries and 
124      * {@code false} otherwise.
125      * 
126      * @param <K>  the key type.
127      * @param map1  the first map.
128      * @param map2  the second map.
129      * 
130      * @return A boolean.
131      */
132     public static <K extends Comparable<K>> boolean equal(Map<K, Paint> map1, Map<K, Paint> map2) {
133         if (!map1.keySet().equals(map2.keySet())) {
134             return false;
135         }
136         for (K key : map1.keySet()) {
137             Paint p1 = map1.get(key);
138             Paint p2 = map2.get(key);
139             if (!PaintUtils.equal(p1, p2)) {
140                 return false;
141             }
142         }
143         return true;
144     }

147      * Converts a color into a string. If the color is equal to one of the
148      * defined constant colors, that name is returned instead. Otherwise the
149      * color is returned as hex-string.
150      *
151      * @param c the color.
152      * @return the string for this color.
153      */
154     public static String colorToString(Color c) {
155         try {
156             Field[] fields = Color.class.getFields();
157             for (int i = 0; i < fields.length; i++) {
158                 Field f = fields[i];
159                 if (Modifier.isPublic(f.getModifiers())
160                         && Modifier.isFinal(f.getModifiers())
161                         && Modifier.isStatic(f.getModifiers())) {
162                     final String name = f.getName();
163                     final Object oColor = f.get(null);
164                     if (oColor instanceof Color) {
165                         if (c.equals(oColor)) {
166                             return name;
167                         }
168                     }
169                 }
170             }
171         } catch (Exception e) {
172             //
173         }
174 
175         // no defined constant color, so this must be a user defined color
176         final String color = Integer.toHexString(c.getRGB() & 0x00ffffff);
177         final StringBuilder retval = new StringBuilder(7);
178         retval.append("#");
179 
180         final int fillUp = 6 - color.length();
181         for (int i = 0; i < fillUp; i++) {
182             retval.append("0");
183         }
184 
185         retval.append(color);
186         return retval.toString();
187     }

189     /**
190      * Converts a given string into a color.
191      *
192      * @param value the string, either a name or a hex-string.
193      * @return the color.
194      */
195     public static Color stringToColor(String value) {
196         if (value == null) {
197             return Color.BLACK;
198         }
199         try {
200             // get color by hex or octal value
201             return Color.decode(value);
202         } catch (NumberFormatException nfe) {
203             // if we can't decode lets try to get it by name
204             try {
205                 // try to get a color by name using reflection
206                 final Field f = Color.class.getField(value);
207                 return (Color) f.get(null);
208             } catch (Exception ce) {
209                 // if we can't get any color return black
210                 return Color.BLACK;
211             }
212         }
213     }
