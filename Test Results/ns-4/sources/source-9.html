177     /**
178      * Calculates and sets the bounds of all the items in the specified
179      * container, subject to the given constraint.  The {@code Graphics2D}
180      * can be used by some items (particularly items containing text) to
181      * calculate sizing parameters.
182      *
183      * @param container  the container whose items are being arranged.
184      * @param constraint  the size constraint.
185      * @param g2  the graphics device.
186      *
187      * @return The container size after the arrangement.
188      */
189     protected Size2D arrangeNF(BlockContainer container, Graphics2D g2,
190                                RectangleConstraint constraint) {
191 
192         List<Block> blocks = container.getBlocks();
193 
194         double height = constraint.getHeight();
195         if (height <= 0.0) {
196             height = Double.POSITIVE_INFINITY;
197         }
198 
199         double x = 0.0;
200         double y = 0.0;
201         double maxWidth = 0.0;
202         List<Block> itemsInColumn = new ArrayList<>();
203         for (Block block : blocks) {
204             Size2D size = block.arrange(g2, RectangleConstraint.NONE);
205             if (y + size.height <= height) {
206                 itemsInColumn.add(block);
207                 block.setBounds(
208                     new Rectangle2D.Double(x, y, size.width, size.height)
209                 );
210                 y = y + size.height + this.verticalGap;
211                 maxWidth = Math.max(maxWidth, size.width);
212             }
213             else {
214                 if (itemsInColumn.isEmpty()) {
215                     // place in this column (truncated) anyway
216                     block.setBounds(
217                         new Rectangle2D.Double(
218                             x, y, size.width, Math.min(size.height, height - y)
219                         )
220                     );
221                     y = 0.0;
222                     x = x + size.width + this.horizontalGap;
223                 }
224                 else {
225                     // start new column
226                     itemsInColumn.clear();
227                     x = x + maxWidth + this.horizontalGap;
228                     y = 0.0;
229                     maxWidth = size.width;
230                     block.setBounds(
231                         new Rectangle2D.Double(
232                             x, y, size.width, Math.min(size.height, height)
233                         )
234                     );
235                     y = size.height + this.verticalGap;
236                     itemsInColumn.add(block);
237                 }
238             }
239         }
240         return new Size2D(x + maxWidth, constraint.getHeight());
241     }
