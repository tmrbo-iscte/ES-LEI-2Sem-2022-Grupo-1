43 /**
44  * Interface for generating the tooltip fragment of an HTML image map area tag.
45  * The fragment should be {@code XHTML 1.0} compliant.
46  */
47 public interface ToolTipTagFragmentGenerator {
48 
49     /**
50      * Generates a tooltip string to go in an HTML image map.  To allow for
51      * varying standards compliance among browsers, this method is expected
52      * to return an 'alt' attribute IN ADDITION TO whatever it does to create
53      * the tooltip (often a 'title' attribute).
54      * <br><br>
55      * Note that the {@code toolTipText} may have been generated from
56      * user-defined data, so care should be taken to filter/escape any
57      * characters that may corrupt the HTML tag.
58      *
59      * @param toolTipText  the tooltip.
60      *
61      * @return The formatted HTML area tag attribute(s).
62      */
63     String generateToolTipFragment(String toolTipText);
