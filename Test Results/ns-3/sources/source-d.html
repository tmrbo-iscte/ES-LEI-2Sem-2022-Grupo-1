40 import java.text.FieldPosition;

43 import org.jfree.chart.internal.Args;

45 /**
46  * A formatter that displays numbers as directions.
47  */
48 public class CompassFormat extends NumberFormat {
49 
50     /** The directions. */
51     public final String[] directions;
52 
53     /**
54      * Creates a new formatter using English identifiers.
55      */
56     public CompassFormat() {
57         this("N", "E", "S", "W");
58     }
59 
60     /**
61      * Creates a new formatter using the specified identifiers for
62      * the base wind directions.
63      * 
64      * @param n  the code for NORTH.
65      * @param e  the code for EAST.
66      * @param s  the code for SOUTH.
67      * @param w  the code for WEST.
68      */
69     public CompassFormat(String n, String e, String s, String w) {
70         this(new String[] {
71             n, n + n + e, n + e, e + n + e, e, e + s + e, s + e, s + s + e, s,
72             s + s + w, s + w, w + s + w, w, w + n + w, n + w, n + n + w
73         });
74     }
75 
76     /**
77      * Creates a new formatter using the specified identifiers.
78      * 
79      * @param directions  an array containing 16 strings representing
80      *     the directions of a compass.
81      */
82     public CompassFormat(String[] directions) {
83         super();
84         Args.nullNotPermitted(directions, "directions");
85         if (directions.length != 16) {
86             throw new IllegalArgumentException("The 'directions' array must "
87                     + "contain exactly 16 elements");
88         }
89         this.directions = directions;
90     }
91 
92     /**
93      * Returns a string representing the direction.
94      *
95      * @param direction  the direction.
96      *
97      * @return A string.
98      */
99     public String getDirectionCode(double direction) {
100         direction = direction % 360;
101         if (direction < 0.0) {
102             direction = direction + 360.0;
103         }
104         int index = ((int) Math.floor(direction / 11.25) + 1) / 2;
105         return directions[index];
106     }
107 
108     /**
109      * Formats a number into the specified string buffer.
110      *
111      * @param number  the number to format.
112      * @param toAppendTo  the string buffer.
113      * @param pos  the field position (ignored here).
114      *
115      * @return The string buffer.
116      */
117     @Override
118     public StringBuffer format(double number, StringBuffer toAppendTo,
119             FieldPosition pos) {
120         return toAppendTo.append(getDirectionCode(number));
121     }
122 
123     /**
124      * Formats a number into the specified string buffer.
125      *
126      * @param number  the number to format.
127      * @param toAppendTo  the string buffer.
128      * @param pos  the field position (ignored here).
129      *
130      * @return The string buffer.
131      */
132     @Override
133     public StringBuffer format(long number, StringBuffer toAppendTo,
134             FieldPosition pos) {
135         return toAppendTo.append(getDirectionCode(number));
136     }

144      *
145      * @return {@code null}.
146      */
147     @Override
148     public Number parse(String source, ParsePosition parsePosition) {
149         return null;
150     }
