46 /**
47  * Generates tooltips using the Dynamic Drive DHTML Tip Message
48  * library (http://www.dynamicdrive.com).
49  */
50 public class DynamicDriveToolTipTagFragmentGenerator
51         implements ToolTipTagFragmentGenerator {
52 
53     /** The title, empty string not to display */
54     protected String title = "";
55 
56     /** The style number */
57     protected int style = 1;
58 
59     /**
60      * Blank constructor.
61      */
62     public DynamicDriveToolTipTagFragmentGenerator() {
63         super();
64     }
65 
66     /**
67      * Creates a new generator with specific title and style settings.
68      *
69      * @param title  title for use in all tooltips, use empty String not to
70      *               display a title.
71      * @param style  style number, see http://www.dynamicdrive.com for more
72      *               information.
73      */
74     public DynamicDriveToolTipTagFragmentGenerator(String title, int style) {
75         this.title = title;
76         this.style = style;
77     }
78 
79     /**
80      * Generates a tooltip string to go in an HTML image map.
81      *
82      * @param toolTipText  the tooltip.
83      *
84      * @return The formatted HTML area tag attribute(s).
85      */
86     @Override
87     public String generateToolTipFragment(String toolTipText) {
88         return " onMouseOver=\"return stm(['"
89             + ImageMapUtils.javascriptEscape(this.title) + "','"
90             + ImageMapUtils.javascriptEscape(toolTipText) + "'],Style["
91             + this.style + "]);\"" + " onMouseOut=\"return htm();\"";
92     }
93 
94 }
