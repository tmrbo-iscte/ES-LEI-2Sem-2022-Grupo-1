520      * Serialises an {@code AttributedString} object.
521      *
522      * @param as  the attributed string object ({@code null} permitted).
523      * @param stream  the output stream ({@code null} not permitted).
524      *
525      * @throws IOException if there is an I/O error.
526      */
527     public static void writeAttributedString(AttributedString as,
528             ObjectOutputStream stream) throws IOException {
529 
530         Args.nullNotPermitted(stream, "stream");
531         if (as != null) {
532             stream.writeBoolean(false);
533             AttributedCharacterIterator aci = as.getIterator();
534             // build a plain string from aci
535             // then write the string
536             StringBuffer plainStr = new StringBuffer();
537             char current = aci.first();
538             while (current != CharacterIterator.DONE) {
539                 plainStr = plainStr.append(current);
540                 current = aci.next();
541             }
542             stream.writeObject(plainStr.toString());
543 
544             // then write the attributes and limits for each run
545             current = aci.first();
546             int begin = aci.getBeginIndex();
547             while (current != CharacterIterator.DONE) {
548                 // write the current character - when the reader sees that this
549                 // is not CharacterIterator.DONE, it will know to read the
550                 // run limits and attributes
551                 stream.writeChar(current);
552 
553                 // now write the limit, adjusted as if beginIndex is zero
554                 int limit = aci.getRunLimit();
555                 stream.writeInt(limit - begin);
556 
557                 // now write the attribute set
558                 Map atts = new HashMap(aci.getAttributes());
559                 stream.writeObject(atts);
560                 current = aci.setIndex(limit);
561             }
562             // write a character that signals to the reader that all runs
563             // are done...
564             stream.writeChar(CharacterIterator.DONE);
565         } else {
566             // write a flag that indicates a null
567             stream.writeBoolean(true);
568         }
569 
570     }
